# 冒泡排序
思路：每轮循环将最大值排到末尾
```bash
let arr = [4, 1, 6, 8, 3, 2, 9, 7];
# 比较之后需要得出是否需要交换
function compare(a, b) {
  return arr[a] > arr[b]
}
# 换位
function exchange(arr, a, b) {
  let temp = arr[a];
  arr[a] = arr[b];
  arr[b] = temp;
}

function sort() {
  for (let i = 0; i < arr.length; i++) {
    for (let j = 0; j < arr.length - 1 - i; j++) {
      if (compare(j, j + 1)) {
        exchange(arr, j, j + 1)
      }
    }
  }

}
```

# 选择排序
思路：选择排序，内层循环，每一圈选出一个最大的，然后放在后面
```bash
let arr = [4, 1, 6, 8, 3, 2, 9, 7];

# 比较之后需要得出是否需要交换
function compare(a, b) {
  return a < b
}
function exchange(arr, a, b) {
  let temp = arr[a];
  arr[a] = arr[b];
  arr[b] = temp;
}

function sort() {
  for (let i = 0; i < arr.length; i++) {
    let maxIndex = 0;
    for (let j = 0; j < arr.length - i; j++) {
      # 每一圈选出一个最大的
      if (compare(arr[maxIndex], arr[j])) {
        maxIndex = j
      }
    }
    exchange(arr, arr.length - 1 - i ,maxIndex)
  }
}
sort()
```

# 简单的快速排序
思想：每循环一次，以第一个数为组长，比组长小的放在左边，比组长小的放在右边，循环之后拼接
```bash
let arr = [4, 1, 6, 8, 3, 2, 9, 7];
function quickSort (arr) {
  # 选班长
  if (arr == null || arr.length == 0) return [];
  # 选班长
  let leader = arr[0]
  # 小的站左边，大的站右边
  let left = [];
  let right = [];
  for (let i = 1; i < arr.length; i++) {
    if (arr[i] < leader) left.push(arr[i])
    else right.push(arr[i])
  }
  return [...quickSort(left), leader, ...quickSort(right)]
}
console.log(quickSort(arr))
```

# 标准快速排序
```bash
let arr = [4, 1, 6, 8, 3, 2, 9, 7];
        # [4, 1, 2, 8, 3, 6, 9, 7]
        # [4, 1, 2, 3, 8, 6, 9, 7]
        # [3, 1, 2, 4, 8, 6, 9, 7]

function quickSort(arr, begin, end) {
  if (begin >= end - 1) return;
  let left = begin;
  let right = end;
  do {
    do left++; while (left < right && arr[left] < arr[begin]);
    do right--; while (right > left && arr[right] > arr[begin])
    if (left < right) {
      exchange(arr, left, right)
    }
  } while (left < right);
  let swapPoin = left == right ? right - 1 : right
  exchange(arr, begin, swapPoin)
  quickSort(arr, begin, swapPoin)
  quickSort(arr, swapPoin + 1, end)
}
function exchange(arr, a, b) {
  let temp = arr[a];
  arr[a] = arr[b];
  arr[b] = temp;
}
quickSort(arr, 0, arr.length);
console.log(arr)
```
